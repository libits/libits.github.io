##  数组中只有一个数出现了奇数次

```go

/*
数组中只有一个数出现了奇数次
*/
package main

import "log"

func main() {
	arr := []int{4, 3, 7, 2, 2, 3, 4, 7, 7}
	ans := oddTimeOneKind(arr)
	log.Println(ans)
}
func oddTimeOneKind(arr []int) int {
	eor := 0
	for i := 0; i < len(arr); i++ {
		eor = eor ^ arr[i]
	}
	return eor
}

```

##  数组中有两个数出现了奇数次

```go
/*
数组中二个数出现了奇数次
*/
package main

import "log"

func main() {
	arr := []int{4, 3, 7, 2, 2, 3, 4, 7, 7, 3}
	ans := oddTime2(arr)
	log.Println(ans)
}
func oddTime2(arr []int) []int {
	eor := 0
	for i := 0; i < len(arr); i++ {
		eor = eor ^ arr[i]
	}
	rightOne := eor & (-eor)
	onlyOne := 0
	for i := 0; i < len(arr); i++ {
		if (arr[i] & rightOne) != 0 {
			onlyOne ^= arr[i]
		}

	}
	ans := make([]int, 2)
	ans[0] = onlyOne
	ans[1] = eor ^ onlyOne
	return ans
}
```
