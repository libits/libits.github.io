##  定义中间件

```go
func Logger() gin.HandlerFunc {
	return func(c *gin.Context) {
		t := time.Now()

		// Set example variable
		c.Set("example", "12345")

		// before request

		c.Next()

		// after request
		latency := time.Since(t)
		log.Print(latency)

		// access the status we are sending
		status := c.Writer.Status()
		log.Println(status)
	}
}

func main() {
	r := gin.New()
	r.Use(Logger())

	r.GET("/test", func(c *gin.Context) {
		example := c.MustGet("example").(string)

		// it would print: "12345"
		log.Println(example)
	})

	// Listen and serve on 0.0.0.0:8080
	r.Run(":8080")
}
```

##  中间件使用

```go
package main

import (
	"fmt"
	"net/http"
	"time"

	"github.com/gin-gonic/gin"
)

var limitChan = make(chan struct{}, 10)

func timeMiddleware() gin.HandlerFunc {
	return func(ctx *gin.Context) {
		begin := time.Now()
		ctx.Next()
		fmt.Printf("use time %d ms\n", time.Since(begin).Milliseconds())
	}
}

func limitMiddleware() gin.HandlerFunc {
	return func(ctx *gin.Context) {
		limitChan <- struct{}{}
		ctx.Next()
		<-limitChan
	}
}

//业务Handler
func bHandler(ctx *gin.Context) {
	time.Sleep(100 * time.Millisecond)
	ctx.String(http.StatusOK, "高性能Golang\n")
}
func main() {
	engine := gin.Default()
	engine.Use(limitMiddleware())
	engine.Use(timeMiddleware())
	engine.GET("/1", bHandler)
	engine.GET("/2", bHandler, bHandler)
	engine.Run("127.0.0.1:5678")
}

```
