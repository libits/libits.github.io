## 接口耗时

计算接口耗时



```go
package main

import (
	"fmt"
	"time"
)

func sort_1(a, b int) string {
	t0 := time.Now()
	defer func() {
        //计算比较时间
		fmt.Printf("耗时%d", time.Since(t0).Milliseconds())
	}()
	if a > b {
		time.Sleep(100 * time.Millisecond)
		return "ok"
	} else {
		time.Sleep(200 * time.Millisecond)
		return "ok"
	}
}
func main() {
	sort_1(2, 3)
}


/*
耗时203
*/
```
