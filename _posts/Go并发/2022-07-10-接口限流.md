## 解释

将协程是100的并发设置成10

```go
package main

import (
	"fmt"
	"sync/atomic"
	"time"
)

var (
	count int32
	//定义一个管道
	count_limit = make(chan struct{}, 10)
)

func readDB() string {
	atomic.AddInt32(&count, 1)
	fmt.Printf("readDB()调用次数%d\n", atomic.LoadInt32(&count))
	atomic.AddInt32(&count, -1)
	return "ok"
}

//定义接口
func i_1() {
	count_limit <- struct{}{}
	readDB()
	<-count_limit
	return
}
func main() {
	for i := 0; i < 100; i++ {
		go i_1()
	}
	time.Sleep(3 * time.Second)
}
```
