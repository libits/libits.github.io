// 题目一：二分查找有序数组中第一次出现数的位置。（c++解题）
public int search (int[] nums, int target) {
       if(nums.length == 0) {
           return -1;
       }
        int begin = 0;
        int end = nums.length - 1;
        while(begin < end) {
            int mid = (begin + end) / 2;
            if(nums[mid] < target) {
                begin = mid + 1;
            } else {
                end = mid; 
            }
        }
        return nums[begin] == target ? begin : -1;
    }
