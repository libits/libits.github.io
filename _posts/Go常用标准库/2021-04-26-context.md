##  context的interface

```go
type Context interface {
   
    Deadline() (deadline time.Time, ok bool)
 
    Done() <-chan struct{}

    Err() error

    Value(key interface{}) interface{}
}
```

##  父-子-Value使用

```go
package main

import (
	"context"
	"fmt"
)

func t_1(c context.Context) context.Context {
    //c为父context，child为子context
	child := context.WithValue(c, "name", "li")
	return child
}
func t_2(c context.Context) context.Context {
	child := context.WithValue(c, "age", "15")
	return child
}
func t_3(c context.Context) {
	fmt.Printf("name %s\n", c.Value("name"))
	fmt.Printf("age %d\n", c.Value("age"))

}
func main() {
	grandpa := context.TODO()
	father := t_1(grandpa)
	grandson := t_2(father)
	t_3(grandson)
}

```

##  Done()函数

```go
package main

import (
	"context"
	"fmt"
	"time"
)

func t_1() {
	c, cancel := context.WithTimeout(context.TODO(), time.Millisecond*1000)
	defer cancel()
	select {
	case <-c.Done():
		err := c.Err()
		fmt.Println(err)
	}
}
func t_2() {
	parent, cancel1 := context.WithTimeout(context.TODO(), time.Millisecond*1000)
	defer cancel1()
	t0 := time.Now()
	time.Sleep(time.Millisecond * 500)

	child, cancel2 := context.WithTimeout(parent, time.Millisecond*100)
	defer cancel2()
	t1 := time.Now()

	select {
	//100ms+500ms约等于600ms
	case <-child.Done():
		err := child.Err()
		t3 := time.Now()
		fmt.Println(t3, t3.Sub(t0).Milliseconds(), t3.Sub(t1).Milliseconds())
		fmt.Println(err)
	}
}
func main() {
	//t_1()
	t_2()
}

/*
620 116
*/
```

```go
package main

import (
	"context"
	"fmt"
	"time"
)

func t_3() {
	c, cancel := context.WithCancel(context.TODO())
	t0 := time.Now()
	go func() {
		time.Sleep(time.Millisecond * 100)
		cancel()
	}()
	select {
	//100ms之后Done管道关闭
	case <-c.Done():
		t3 := time.Now()
		fmt.Println(t3.Sub(t0).Milliseconds())
		err := c.Err()
		fmt.Println(err)
	}

}
func main() {
	t_3()
}
/*
114
*/
```
